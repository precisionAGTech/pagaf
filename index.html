example.py 
from nltgis import Client

# Connect to nltGIS.ai
client = Client(api_key="your-key")

# Ask about your field
result = client.ask(
    "NDVI analysis for my corn field",
    geojson="my_field.geojson"
)

# Get results
result.to_map()      # Interactive map
result.to_geojson()  # GeoJSON
result.to_geotiff()  # Prescription map
